;Calculating frames from timestamps assuming 60 FPS
;Functions:
;frames - Input 2 lists of 2 numbers
;loadless - Input a list of 2 number element lists

(define frames
  (lambda (start stop)
    (+ (* (- (first stop) (first start)) 60)
       (- (second stop) (second start))
       1)))

(define loadless
  (lambda (listy)
    (list (frame-sum listy)
          (frames (first listy)
                  (last listy)))))

(define frame-sum
  (lambda (listy)
    (cond
      ((null? (rest (rest listy)))
       (frames (first listy)
               (second listy)))
      (#t
       (+ (frames (first listy)
                  (second listy))
          (frame-sum (rest (rest listy))))))))
